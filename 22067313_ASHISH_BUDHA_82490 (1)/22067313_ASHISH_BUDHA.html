from scipy.stats import skew

summary_stats = df.groupby('job_title')['salary_in_usd'].describe()
skewness = df.groupby('job_title')['salary_in_usd'].apply(lambda x: skew(x))

summary_stats['skewness'] = skewness
kurtosis = df.groupby('job_title')['salary_in_usd'].apply(pd.DataFrame.kurt)
summary_stats['kurtosis'] = kurtosis
print(summary_stats)

data_scientist = df[df['job_title'] == 'Data Scientist']

minSalary = data_scientist['salary_in_usd'].min()
minSalary = (minSalary // 100) * 100
maxSalary = data_scientist['salary_in_usd'].max()
maxSalary = (maxSalary // 100) * 100
interval = 20000
nums = int((maxSalary - minSalary) / interval) + 1
plot.figure(figsize = (10,6))
plot.hist(data_scientist['salary_in_usd'], bins = nums)

plot.xlabel('Salary')
plot.ylabel('Number of people')
plot.xticks(range(minSalary, maxSalary + interval, interval), rotation = 45)
plot.tight_layout()
plot.show()
